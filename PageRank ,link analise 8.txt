import numpy as np


def pagerank(graph, damping_factor=0.85, max_iterations=100, epsilon=1e-8):

    num_pages = graph.shape[0]

    
    page_rank = np.ones(num_pages) / num_pages

    
    out_degree = np.sum(graph, axis=1)

    
    teleport = (1 - damping_factor) / num_pages

    for _ in range(max_iterations):

        prev_page_rank = page_rank.copy()

        for i in range(num_pages):

            
            incoming_links = np.where(graph[:, i] == 1)[0]

            rank_sum = 0

            for j in incoming_links:
                if out_degree[j] > 0:
                    rank_sum += prev_page_rank[j] / out_degree[j]

            page_rank[i] = teleport + damping_factor * rank_sum

        
        if np.linalg.norm(page_rank - prev_page_rank, 1) < epsilon:
            break

    return page_rank


if __name__ == "__main__":

    
    web_graph = np.array([
        [0, 1, 1, 1],
        [1, 0, 0, 0],
        [1, 1, 0, 1],
        [0, 0, 1, 0]
    ])

    page_rank_scores = pagerank(web_graph)

    print("PageRank Scores:\n")
    for i, score in enumerate(page_rank_scores):
        print(f"Page {i + 1}: {score:.4f}")
